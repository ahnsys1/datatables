-- Drop the function if it exists to ensure a clean recreation
DROP FUNCTION IF EXISTS public.insert_dummy_users(INT);

-- Create a function to insert a specified number of dummy users
CREATE OR REPLACE FUNCTION public.insert_dummy_users(
    num_to_insert INT
)
    RETURNS void
    LANGUAGE 'plpgsql'
AS $BODY$
DECLARE
    i INT;
    first_names TEXT[] := ARRAY['Olivia', 'Liam', 'Emma', 'Noah', 'Amelia', 'Oliver', 'Ava', 'Elijah', 'Sophia', 'Mateo'];
    last_names TEXT[] := ARRAY['Smith', 'Johnson', 'Williams', 'Brown', 'Jones', 'Garcia', 'Miller', 'Davis', 'Rodriguez', 'Martinez'];
    selected_first_name TEXT;
    selected_last_name TEXT;
    generated_username TEXT;
    generated_email TEXT;
    generated_age INT;
BEGIN
    -- Loop for the specified number of times
    FOR i IN 1..num_to_insert LOOP
        -- Select a random first and last name from the arrays
        selected_first_name := first_names[1 + floor(random() * array_length(first_names, 1))];
        selected_last_name := last_names[1 + floor(random() * array_length(last_names, 1))];

        -- Generate username: prefix + lowercase(firstname) + lowercase(lastname)
        generated_username := i || '_' || lower(selected_first_name) || '_' || lower(selected_last_name);

        -- Generate a corresponding email address
        generated_email := generated_username || '@example.com';

        -- Generate a random age between 18 and 65
        generated_age := floor(random() * (65 - 18 + 1) + 18);

        INSERT INTO public.users(
            id,
            username,
            email,
            first_name,
            last_name,
            age
        )
        VALUES (
            gen_random_uuid()::varchar,
            generated_username,
            generated_email,
            selected_first_name,
            selected_last_name,
            generated_age
        );
    END LOOP;
END;
$BODY$;

ALTER FUNCTION public.insert_dummy_users(INT)
    OWNER TO postgres;
